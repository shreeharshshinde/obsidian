We have to precompute all four pillars for checking if the matrix is Magic Matrix.
1. Row Prefix Sum
2. Column Prefix Sum 
3. Main Diagonal Sum 
4. Opposite Diagonal Sum

then for, every k : from min(m, n) to 2 => check for every (i, j) in matrix

```cpp
int largestMagicMatrix(vector<vectpr<int>>& grid) {
	int m = grid.size();
	int n = grid[0].size();
	
	vector<vector<int>> rowPref(m+1, vector<int>(n+1, 0));
	for(int i=1; i<=m; i++) {
		for(int j=1; j<=n; j++) {
			rowPref[i][j] = grid[i-1][j-1] + rowPref[i][j-1];
		}
	}
	
	vector<vector<int>> colPref(m+1, vector<int>(n+1, 0));
	for(int i=1; i<=m; i++) {
		for(int j=1; j<=n; j++) {
			colPref[i][j] = grid[i-1][j-1] + colPref[i-1][j];
		}
	}
	
}
```
